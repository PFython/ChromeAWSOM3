import json
from js import document
from js import getChrome, clipboardCopy
from js import insertContentScript  # insert and run content.js (1st pass)

# DISPLAY ACTIVE BROWSER TAB METADATA

chrome = getChrome()
tabs = await chrome.tabs.query(["https://*/*"])
active_tab = [x for x in tabs if x.active][0]
metadata_as_dict = active_tab.to_py()
metadata_as_json = json.dumps(metadata_as_dict, indent=2)
metadata = document.getElementById('metadata')
metadata.innerHTML = f"{metadata_as_json}."

# RETRIEVE ACTIVE BROWSER TAB DOM AS JSON

insertContentScript(active_tab.id)  # insert and run content.js (2nd pass)
