import json
from js import document, chrome

# DISPLAY ACTIVE BROWSER TAB METADATA
tabs = await chrome.tabs.query(["http*://*/*"])
active_tab = [x for x in tabs if x.active][0]
metadata_as_dict = active_tab.to_py()
metadata_as_json = json.dumps(metadata_as_dict, indent=2)
insertion_point = document.getElementById('metadata')
insertion_point.innerHTML = f"{metadata_as_json}."
